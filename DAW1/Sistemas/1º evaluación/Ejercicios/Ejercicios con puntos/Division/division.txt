;B0 DIVIDENDO
;B1 DIVISOR
;B2 SUMA COCIENTE
;B3 RESTO

LDA #5
STA $B0
LDA #3
STA $B1
LDA #0
STA $B2

LDA $B0;  	CARGAMOS CANTIDAD DIVIDENDO
CMP $B1; 	COMPARAMOS CON DIVISOR
SUB $B1; 	RESTAMOS DIVISOR AL DIVIDENDO
JNC $1C;	SALTO AL RESTO EN CASO DIVIDENDO<DIVISOR
STA $B0;	ACTUALIZAMOS DIVIDENDO
LDA $B2; 	CARGAMOS SUMA
INA; 	        INCREMENTAMOS COCIENTE
STA $B2;	ACTUALIZAMOS SUMA
JMP $0C;	SALTO 1ER PASO

LDA $B0;	CARGAMOS DIVIDENDO
STA $B3;	GUARDAMOS RESTO

HLT;
