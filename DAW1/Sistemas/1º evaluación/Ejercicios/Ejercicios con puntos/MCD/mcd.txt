LDA #0
LDA $FF;	D%d/RESTO
LDA #8
STA $FD;	DIVIDENDO(D)
LDA #4
STA $FE;	DIVISOR(d)/MCD CUANDO HLT

CMP #0;		DESCARTE CASO TRIVIAL INICIAL(DIVISOR=0)
JNZ $14
LDA $FD
STA $FE
HLT;		FIN CASO TRIVIAL

LDA $FD;	$14.(INICIO D%d)CARGAMOS CANTIDAD DIVIDENDO
CMP $FE;	COMPARAMOS CON DIVISOR
JNC $1F;	SALTO AL RESTO EN CASO DIVIDENDO<DIVISOR
SUB $FE; 	RESTAMOS DIVISOR AL DIVIDENDO
STA $FD;	ACTUALIZAMOS DIVIDENDO
JMP $14;	SALTO AL 1ER PASO (REPETIREMOS HASTA QUE DIVIDENDO<DIVISOR)

STA $FF;	$1F.GUARDAMOS RESTO(FIN D%d)

CMP #0;		COMPARAMOS SI EL RESTO ES CERO
JZ $2E;		EN CASO DE SERLO HEMOS ACABADO PORQUE ES UN CASO TRIVIAL Y EL MCD ES EL DIVISOR
LDA $FE;	CARGAMOS DIVISOR
STA $FD;	GUARDAMOS EN DIVIDENDO
LDA $FF;	CARGAMOS RESTO
STA $FE;	GUARDAMOS EN DIVISOR
JMP $14;	REPETIMOS HASTA LLEGAR A UN CASO TRIVIAL

HLT
